---
title: "Get schedule"
description: "Scrapes class schedule for all semesters concurrently from i-Ma'luum."
api: "GET https://api.quddus.my/api/schedule"
---

Fetches class timetable data for every available semester. Sessions are processed concurrently using a goroutine worker pool and the response is sorted by session name in descending order.

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from the login endpoint. Format: `Bearer <token>`
</ParamField>

<ParamField header="x-gomaluum-key" type="string">
  Optional API key for an additional security layer. Required when the server is configured to enforce API key authentication.
</ParamField>

## Response

<ResponseField name="message" type="string">
  Human-readable status message. Value is `Successfully fetched schedule` on success.
</ResponseField>

<ResponseField name="data" type="ScheduleResponse[]">
  Array of schedule objects, one per semester. Sorted by session name.
  <Expandable title="ScheduleResponse properties" defaultOpen>
    <ResponseField name="id" type="string">
      Unique identifier for this schedule entry, prefixed with `gomaluum:schedule:`.
    </ResponseField>
    <ResponseField name="session_name" type="string">
      Human-readable semester name (e.g. `2023/2024 - 1`).
    </ResponseField>
    <ResponseField name="session_query" type="string">
      URL query string used to retrieve this semester's data from i-Ma'luum (e.g. `?ses=2023/2024&sem=1`).
    </ResponseField>
    <ResponseField name="schedule" type="ScheduleSubject[]">
      List of subjects enrolled in this semester.
      <Expandable title="ScheduleSubject properties">
        <ResponseField name="id" type="string">
          Unique identifier for this subject entry, prefixed with `gomaluum:subject:`.
        </ResponseField>
        <ResponseField name="course_code" type="string">
          Official course code (e.g. `INFO 2301`).
        </ResponseField>
        <ResponseField name="course_name" type="string">
          Full course title (e.g. `Web Application Development`).
        </ResponseField>
        <ResponseField name="section" type="number">
          Section number the student is registered in.
        </ResponseField>
        <ResponseField name="chr" type="number">
          Credit hours for this course.
        </ResponseField>
        <ResponseField name="venue" type="string">
          Room or building where the class is held (e.g. `E2.2-RM1`).
        </ResponseField>
        <ResponseField name="lecturer" type="string">
          Name of the lecturer assigned to this section.
        </ResponseField>
        <ResponseField name="timestamps" type="WeekTime[]">
          Array of weekly meeting times. A course with multiple days will have one entry per day.
          <Expandable title="WeekTime properties">
            <ResponseField name="start" type="string">
              Start time in `HHMM` 24-hour format (e.g. `0800`).
            </ResponseField>
            <ResponseField name="start_unix" type="number">
              Unix timestamp of the start time computed for today's date in the `Asia/Kuala_Lumpur` timezone. Omitted if parsing fails.
            </ResponseField>
            <ResponseField name="end" type="string">
              End time in `HHMM` 24-hour format (e.g. `1000`).
            </ResponseField>
            <ResponseField name="end_unix" type="number">
              Unix timestamp of the end time computed for today's date in the `Asia/Kuala_Lumpur` timezone. Omitted if parsing fails.
            </ResponseField>
            <ResponseField name="day" type="number">
              Day of the week as an integer. `1` = Monday, `2` = Tuesday, `3` = Wednesday, `4` = Thursday, `5` = Friday, `6` = Saturday, `0` = Sunday.
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

  },
});
