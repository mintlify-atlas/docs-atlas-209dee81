---
title: "Get result"
description: "Scrapes academic results for all semesters concurrently from i-Ma'luum."
api: "GET https://api.quddus.my/api/result"
---

Fetches examination results for every available semester. Sessions are processed concurrently using a goroutine worker pool and the response is sorted by session name in descending order.

## Request

<ParamField header="Authorization" type="string" required>
  Bearer token obtained from the login endpoint. Format: `Bearer <token>`
</ParamField>

<ParamField header="x-gomaluum-key" type="string">
  Optional API key for an additional security layer. Required when the server is configured to enforce API key authentication.
</ParamField>

## Response

<ResponseField name="message" type="string">
  Human-readable status message. Value is `Successfully fetched results` on success.
</ResponseField>

<ResponseField name="data" type="ResultResponse[]">
  Array of result objects, one per semester. Sorted by session name.
  <Expandable title="ResultResponse properties" defaultOpen>
    <ResponseField name="id" type="string">
      Unique identifier for this result entry, prefixed with `gomaluum:result:`.
    </ResponseField>
    <ResponseField name="session_name" type="string">
      Human-readable semester name (e.g. `2023/2024 - 1`).
    </ResponseField>
    <ResponseField name="session_query" type="string">
      URL query string used to retrieve this semester's results from i-Ma'luum (e.g. `?ses=2023/2024&sem=1`).
    </ResponseField>
    <ResponseField name="gpa_value" type="string">
      GPA for this semester as a string (e.g. `3.75`).
    </ResponseField>
    <ResponseField name="cgpa_value" type="string">
      Cumulative GPA up to and including this semester as a string (e.g. `3.68`).
    </ResponseField>
    <ResponseField name="credit_hours" type="string">
      Total credit hours attempted in this semester as a string (e.g. `17`).
    </ResponseField>
    <ResponseField name="status" type="string">
      Academic standing for this semester (e.g. `PASS`, `DEAN'S LIST`).
    </ResponseField>
    <ResponseField name="result" type="Result[]">
      List of individual course results for this semester.
      <Expandable title="Result properties">
        <ResponseField name="id" type="string">
          Unique identifier for this course result, prefixed with `gomaluum:subject:`.
        </ResponseField>
        <ResponseField name="course_code" type="string">
          Official course code (e.g. `INFO 2301`).
        </ResponseField>
        <ResponseField name="course_name" type="string">
          Full course title (e.g. `Web Application Development`).
        </ResponseField>
        <ResponseField name="course_grade" type="string">
          Letter grade awarded for this course (e.g. `A`, `B+`, `C`).
        </ResponseField>
        <ResponseField name="course_credit" type="string">
          Credit hours for this course as a string (e.g. `3`).
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
